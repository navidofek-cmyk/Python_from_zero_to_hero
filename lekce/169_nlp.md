## 🌿 Dependency parsing

```python
print("\n=== Závislosti (první věta) ===")
prvni_veta = list(doc.sents)[0]
for token in prvni_veta:
    print(f"  {token.text:<12} ←[{token.dep_}]— {token.head.text}")

# Extrakce subjekt-sloveso-objekt
def extrahuj_svo(doc):
    trojice = []
    for token in doc:
        if token.dep_ == "nsubj":
            subj = token.text
            sloveso = token.head.text
            obj = next((t.text for t in token.head.children if t.dep_ in ("dobj","attr")), None)
            if obj:
                trojice.append((subj, sloveso, obj))
    return trojice

print("\n=== Subjekt-Sloveso-Objekt ===")
for s, v, o in extrahuj_svo(doc):
    print(f"  {s} → {v} → {o}")
```

## 📊 Text preprocessing pipeline

```python
def preprocessuj(text: str, nlp) -> list[str]:
    """Tokenizace, lemmatizace, odstranění stop slov a interpunkce."""
    doc = nlp(text.lower())
    return [
        token.lemma_
        for token in doc
        if not token.is_stop
        and not token.is_punct
        and not token.is_space
        and len(token.text) > 2
    ]


texty = [
    "Python is the best programming language for data science",
    "Machine learning with Python is very popular",
    "I love coding in Python every day",
]

for t in texty:
    zpracovany = preprocessuj(t, nlp)
    print(f"  Původní: {t[:40]}...")
    print(f"  Tokeny:  {zpracovany}\n")
```

## 🔢 TF-IDF a BM25

```python
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np


def tf_idf_vyhledavani(dokumenty: list[str], dotaz: str, top_n: int = 3) -> list[tuple]:
    """TF-IDF vyhledávání v dokumentech."""
    vectorizer = TfidfVectorizer(
        ngram_range=(1, 2),
        min_df=1,
        stop_words="english",
    )
    tfidf_matrix = vectorizer.fit_transform(dokumenty)
    dotaz_vec = vectorizer.transform([dotaz])

    skore = cosine_similarity(dotaz_vec, tfidf_matrix)[0]
    nejlepsi = np.argsort(skore)[::-1][:top_n]

    return [(dokumenty[i], skore[i]) for i in nejlepsi if skore[i] > 0]


dokumenty = [
    "Python is great for machine learning and AI",
    "FastAPI helps build REST APIs quickly",
    "Docker containerizes applications for deployment",
    "Neural networks learn from training data",
    "Python supports object-oriented programming",
]

vysledky = tf_idf_vyhledavani(dokumenty, "Python programming")
print("\n=== TF-IDF Vyhledávání ===")
for dok, skore in vysledky:
    print(f"  [{skore:.3f}] {dok}")
```

## 📝 Vlastní NER trénink

```python
import spacy
from spacy.training import Example
import random


def trenuj_ner(trenovaci_data: list, n_iter: int = 10):
    """
    Natrénuj vlastní NER model.
    trenovaci_data = [("text", {"entities": [(start, end, label)]}), ...]
    """
    nlp = spacy.blank("cs")
    ner = nlp.add_pipe("ner")

    for _, annotations in trenovaci_data:
        for _, _, label in annotations.get("entities", []):
            ner.add_label(label)

    optimizer = nlp.begin_training()
    for iterace in range(n_iter):
        random.shuffle(trenovaci_data)
        ztrata = {}
        for text, annotations in trenovaci_data:
            doc = nlp.make_doc(text)
            example = Example.from_dict(doc, annotations)
            nlp.update([example], drop=0.35, losses=ztrata)
        if iterace % 5 == 0:
            print(f"  Iterace {iterace}: NER ztráta = {ztrata.get('ner', 0):.4f}")

    return nlp


# Ukázková trénovací data (česká místa)
trenovaci = [
    ("Praha je hlavní město České republiky.",
     {"entities": [(0, 5, "LOC"), (25, 44, "GPE")]}),
    ("Brno leží na Moravě.",
     {"entities": [(0, 4, "LOC"), (13, 19, "LOC")]}),
]

print("\n=== Trénink vlastního NER ===")
model = trenuj_ner(trenovaci, n_iter=10)
doc = model("Ostrava je průmyslové město.")
for ent in doc.ents:
    print(f"  {ent.text}: {ent.label_}")
```
